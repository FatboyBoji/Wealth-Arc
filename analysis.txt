```typescript
interface DeviceSession {
    id: string;
    deviceName: string;
    deviceType: string;
    browser: string;
    os: string;
    lastActive: string;
    isCurrentSession: boolean;
}

interface MaxSessionsError {
    sessions: DeviceSession[];
    userId: number;
    message: string;
}

class AuthService {
    async login(credentials: LoginCredentials): Promise<AuthResponse> {
        try {
            // ... existing login logic
        } catch (error: unknown) {
            if (axios.isAxiosError(error)) {
                if (error.response?.status === 400 && 
                    error.response.data?.error === 'MAX_SESSIONS_REACHED') {
                    const errorData = error.response.data;
                    throw new MaxSessionsError(
                        errorData.sessions,
                        errorData.userId,
                        errorData.message || 'Maximum number of sessions reached'
                    );
                }
            }
            throw new ApiError(
                error instanceof Error ? error.message : 'Login failed'
            );
        }
    }

    async getActiveSessions(): Promise<DeviceSession[]>;
    async terminateSession(sessionId: string): Promise<void>;
    async terminateAllOtherSessions(): Promise<void>;
}
```
